file(GLOB_RECURSE SOURCES LIST_DIRECTORIES true *.h *.cpp *.ui *.qrc)

#add_executable(${PROJECT_NAME} ${SOURCES})

if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(${PROJECT_NAME}

    )

            ${SOURCES}
        )
endif()
